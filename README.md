## Install

```bash
npm install --save use-delay
```

## Usage

```tsx
import * as React from 'react'

import { useMyHook } from 'use-delay'

const Example = () => {
  const example = useMyHook()
  return (
    <div>
      {example}
    </div>
  )
}
```
